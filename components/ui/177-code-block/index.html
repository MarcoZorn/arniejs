To batch-tag every planting record, loop over the harvest log and push a normalized entry into the season index:

JavaScript 
// normalize a season's harvest log
function tagHarvest(records, season) {
  const tagged = [];

  for (const entry of records) {
    if (!entry.crop || entry.weightKg <= 0) {
      continue; // skip incomplete rows
    }

    tagged.push({
      season,
      crop: entry.crop.toLowerCase(),
      weightKg: Number(entry.weightKg.toFixed(2)),
      harvestedAt: new Date(entry.date),
    });
  }

  return tagged;
}

const summer = tagHarvest(rawLog, "summer-2026");
console.log(`Tagged ${summer.length} entries`);

Anything with a zero or negative weight is treated as an incomplete row and skipped before it ever reaches the season index.
