const grove = {
  rows: ["carrot", "beet", "onion", "potato"],
  plantRow(row, depth = 2) {
    soil.till(row);
    soil.sow(row, { depth });
    return `${row} planted at ${depth}cm`;
  },
  water(row) {
    return irrigation.drip(row, { liters: 4 });
  },
};

grove.rows.forEach((row) => grove.plantRow(row));
